# 项目克隆

git clone https://github.com/HokageYeah/HokageYeah.github.io

# 运行 打包

# 安装依赖
pnpm install 

# 启动
pnpm run dev

# 打包
pnpm run build

# 发布
pnpm run deploy


# 安装依赖
yarn install 

# 启动
yarn dev

# 打包
yarn  build

# 发布
yarn deploy

npm install 

# 启动
npm run dev

# 打包
npm run build

# 发布
